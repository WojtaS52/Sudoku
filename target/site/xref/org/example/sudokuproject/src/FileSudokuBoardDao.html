3   import java.io.FileInputStream;
4   import java.io.FileOutputStream;
5   import java.io.IOException;
6   import java.io.ObjectInputStream;
7   import java.io.ObjectOutputStream;
8   import org.apache.log4j.Logger;
9   import org.example.sudokuproject.exceptions.DaoException;
10  import org.example.sudokuproject.exceptions.RunExeption;
11  
12  
13  public class FileSudokuBoardDao implements Dao<SudokuBoard>, AutoCloseable {
14  
15      private String f1leName;
16      private static final Logger logger = Logger.getLogger(SudokuBoard.class.getName());
17  
18      public FileSudokuBoardDao(String f1leName) {
19          this.f1leName = f1leName + ".txt";
20      }
21  
22      @Override
23      public final SudokuBoard read() throws DaoException {
24          try (FileInputStream file = new FileInputStream(f1leName);
25               ObjectInputStream obj = new ObjectInputStream(file)) {
26              SudokuBoard board = (SudokuBoard) obj.readObject();
27              return board;
28          } catch (IOException | ClassNotFoundException e) {
29              try {
30                  logger.error("Dao exception while 'read'");
31                  throw new DaoException(e);
32              } catch (DaoException ex) {
33                  logger.error("Dao runtime exception while 'read' ");
34                  throw new RunExeption(ex);
35              }
36          }
37      }
38  
39      @Override
40      public final void write(SudokuBoard x) throws DaoException {
41          try (FileOutputStream file = new FileOutputStream(f1leName);
42               ObjectOutputStream obj = new ObjectOutputStream(file)) {
43              obj.writeObject(x);
44          } catch (IOException e) {
45              try {
46                  logger.error("Dao exception while 'write'");
47                  throw new DaoException(e);
48              } catch (DaoException ex) {
49                  logger.error("Dao runtime exception while 'write'");
50                  throw new RunExeption(ex);
51              }
52          }
53      }
54  
55      public void close() throws DaoException {
56  
57      }
58  }
